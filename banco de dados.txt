CREATE TABLE cliente(
    	nome_clie varchar(30) not null,
	email_clie varchar(30) not null,
	idade_clie date not null,
	tel_fixo_clie VARCHAR(10) not null,
	tel_celular_clie VARCHAR(11) not null,
	senha_clie VARCHAR(10) not null,
	endereço_clie VARCHAR(30) not null,
	complemento_clie varchar(30) null,
	cidade_clie varchar(30) not null,
	estado_clie varchar(30) not null,
	cep_clie varchar(30) not null,
	CPF_clie VARCHAR(11) PRIMARY KEY,
	img_perfil longblob
);

CREATE TABLE funcionario(
    	nome_func varchar(30) not null,
	CPF_func VARCHAR(11) PRIMARY KEY,
    	endereco_func varchar(30) not null,
	complemento_func varchar(30) null,
	idade_func date not null,
	cidade_func varchar(30) not null,
	estado_func varchar(30) not null,
    	tel_func VARCHAR(10) not null,
	instrumento VARCHAR(15) not null,
	senha_func VARCHAR(50) not null,
	cep_func varchar(30) not null,
	email_func VARCHAR(50) not null,
	estilo_musical VARCHAR(15) not null,
	img_perfil longblob,
	disp_func VARCHAR(15)
);

CREATE TABLE realiza_pedido(
	data_hora_evento DATETIME,
	endereço_evento VARCHAR(30),
	estilo_musica_pedido VARCHAR(20),
	metodo_pagamento VARCHAR(15),
	instrumento_desejado VARCHAR(15),
	status_pedido VARCHAR(2),
	desc_pedido varchar(200),
	valor_pedido varchar(50),
	CPF_clie_FK VARCHAR(10),
	CPF_func_FK VARCHAR(10),
	ID_pedido_PK int(11) AUTO_INCREMENT,
    	PRIMARY KEY(ID_pedido_pk),
	FOREIGN KEY(CPF_clie_FK) REFERENCES cliente (CPF_clie),
	FOREIGN KEY(CPF_func_FK) REFERENCES funcionario (CPF_func)
);
